#!/bin/bash
LED OFF

# Arrow to button input map
map_arrow() {
  case "$1" in
    "↑") echo "switch-right" ;;
    "↓") echo "switch-left" ;;
    "←") echo "switch-middle" ;;
  esac
}

arrows=("↑" "↓" "←")
ROUND=1

# Round delay and pattern length
get_round_config() {
  case $1 in
    1) delay=5; length=4; TAUNT="" ;;
    2) delay=4; length=5; TAUNT=":)" ;;
    3) delay=3; length=6; TAUNT="Ok!" ;;
    4) delay=6; length=7; TAUNT="Nice!" ;;
    5) delay=5; length=7; TAUNT="Niiice!" ;;
    6) delay=4; length=7; TAUNT="Good job!" ;;
    7) delay=3; length=7; TAUNT="Well done!" ;;
    8) delay=7; length=8; TAUNT="Way to go!" ;;
    9) delay=6; length=8; TAUNT="Good job!" ;;
    10) delay=5; length=8; TAUNT="On a roll!" ;;
    11) delay=4; length=8; TAUNT="Great!" ;;
    12) delay=3; length=8; TAUNT="Impressive!" ;;
    13) delay=7; length=9; TAUNT="Hammerhead?!" ;;
    14) delay=6; length=9; TAUNT="Rampage!" ;;
    15) delay=5; length=9; TAUNT="Dominating!" ;;
    16) delay=4; length=9; TAUNT="Unstoppable!" ;;
    17) delay=3; length=9; TAUNT="M-M-M-MONSTER!" ;;
    18) delay=7; length=10; TAUNT="MASSACRE!" ;;
    19) delay=6; length=10; TAUNT="Wicked Sick!" ;;
    20) delay=5; length=10; TAUNT="Extermination!" ;;
    21) delay=4; length=10; TAUNT="Killtacular!" ;;
    22) delay=3; length=10; TAUNT="Killpocalypse!" ;;
    23) delay=6; length=11; TAUNT="Combo Breaker!" ;;
    24) delay=5; length=11; TAUNT="ULTRA COMBO!!" ;;
    25) delay=4; length=11; TAUNT="K-Rad WaReZ?!" ;;
    26) delay=3; length=11; TAUNT="uLtRa eLiTe" ;;
    27) delay=5; length=12; TAUNT="ZZOOOMMMMGGGG!" ;;
    28) delay=4; length=12; TAUNT="Phenomenal" ;;
    29) delay=3; length=12; TAUNT="World-Class" ;;
    30) delay=5; length=13; TAUNT="Legendary!" ;;
    31) delay=4; length=13; TAUNT="Next-Level!" ;;
    32) delay=3; length=13; TAUNT="On FIRE!" ;;
    33) delay=5; length=14; TAUNT="==AwEsOmE==" ;;
    34) delay=4; length=14; TAUNT="Remarkable!!" ;;
    35) delay=3; length=14; TAUNT="Outstanding!" ;;
    36) delay=5; length=15; TAUNT="Mind Blowing!" ;;
    37) delay=4; length=15; TAUNT="Daaaaayum!" ;;
    38) delay=3; length=15; TAUNT="EEF5204D6A" ;;
    39) delay=5; length=16; TAUNT="KEEP ROCKIN'" ;;
    40) delay=4; length=16; TAUNT="Go for it!" ;;
    41) delay=3; length=16; TAUNT="DAS KOOL!" ;;
    *) delay=3; length=16; TAUNT="Godlike!" ;;  # Round 42+
  esac
}

while true; do
  # Get config for this round
  get_round_config "$ROUND"

  # Show round & countdown
  ALERT "$TAUNT \n Round $ROUND" 1 false

  # Reset LED and countdown
  LED OFF
  ALERT "3..." 1 false
  ALERT "2..." 1 false
  ALERT "1..." 1 false

  # Generate pattern of $length random arrows
  pattern=()
  for i in $(seq 1 "$length"); do
    rand=$((RANDOM % 3))
    pattern+=("${arrows[$rand]}")
  done

  # Show pattern for $delay seconds
  ALERT "$(printf "%s" "${pattern[@]}")" "$delay" false
  CLEAR_SCREEN

  # Input loop
  correct=true
  for expected in "${pattern[@]}"; do
    input=$(WAIT_FOR_INPUT)
    expected_input=$(map_arrow "$expected")

    if [ "$input" != "$expected_input" ]; then
      correct=false
      break
    fi
  done

  # Success or fail
  if $correct; then
    LED FINISH
    ROUND=$((ROUND + 1))
  else
    LED FAIL
    ALERT "Game Over!\n Score: $((ROUND - 1))" -1 false
    RESP="$(OPTION_DIALOG "Play again?" " Yes " " No ")"
    case "$RESP" in
      " Yes ") ROUND=1 ;;
      " No ") exit ;;
    esac
  fi
done
